import { Button, Card } from "@versini/ui-components";

/**
 * Now that the required components are

 */
function App() {
	return (
		<Card
			header="Welcome to UI Components"
			footer={

					Hooray
				</Button>
			}
		>
			<p>Hello World</p>
		</Card>
	);
}
